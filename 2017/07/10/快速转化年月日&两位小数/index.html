快速转化年月日&两位小数

function timeControl(time){
    var date = new Date(time);
            Y = date.getFullYear() + '-';
            M = (date.getMonth()+1 < 10 ? '0'+(date.getMonth()+1) : date.getMonth()+1) + '-';
            D = date.getDate() + ' ';
            h = (date.getHours()+1 <10 ?'0' + date.getHours():date.getHours())+':';      
            m = (date.getMinutes()+1<10?'0' + date.getMinutes():date.getMinutes())+':';
            s = (date.getSeconds()+1<10?'0'+date.getSeconds():date.getSeconds());
            timeCon = Y+M+D+h+m+s;
            return timeCon;
        }

function fixedTwo(num){
   var m = parseFloat(num)/100;
    if(m >= 0){
        return m.toFixed(2);
    }
    return '0.00';
}

function getFloatStr(num){
        num += '';
        num = num.replace(/[^0-9|\.]/g, ''); //清除字符串中的非数字非.字符
        if(/^0+/) //清除字符串开头的0
            num = num.replace(/^0+/, '');
        if(!/\./.test(num)) //为整数字符串在末尾添加.00
            num += '.00';
        if(/^\./.test(num)) //字符以.开头时,在开头添加0
            num = '0' + num;
        num += '00';        //在字符串末尾补零
        num = num.match(/\d+\.\d{2}/)[0];
        return num;
    }
